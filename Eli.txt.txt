facit = 'Zlatan'

def rätt_spelare(gissning):
    if gissning == facit:
        print('Rätt spelare!')
        return True
    else:
        print('fel, gissa igen!')
        return False
while True:
    gissning = input('Gissa fotbollspelaren jag tänker på!: ')
    if rätt_spelare(gissning) == True:
        break
